##Parameters
 - zipCode: number 
   - format: 00000000

##Address Output

      description: string
      id: number
    }
- city: {

    }
- country: {
      description: string
      id: number
    }
- complement: string
- federationUnity: {
      description: string
      acronym: string
    }
- neighborhood: string
- publicPlace: string
